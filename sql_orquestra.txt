CREATE TABLE Alunos (
rua varchar(50),
cidade varchar(50),
uf varchar(2),
bairro varchar(50),
responsavel_legal varchar(50),
telefone varchar(10),
numero varchar(5),
nome varchar(50),
cpf varchar(11),
cep varchar(10),
id integer PRIMARY KEY
);


CREATE TABLE Turma (
id integer PRIMARY KEY,
nome varchar(50),
horario time,
vigencia varchar(50),
professor_id varchar(50),
turma_id integer,
FOREIGN KEY (turma_id)  REFERENCES Alunos (id)
);

CREATE TABLE Professor (
id integer PRIMARY KEY,
nome varchar(50),
telefone varchar(50),
cpf varchar(50),
turma_id integer,
FOREIGN KEY(turma_id) REFERENCES Turma (id)
);

CREATE TABLE Curso (
nome varchar(50),
professor_id integer,
modalidade varchar(50),
disciplina varchar(50),
id integer PRIMARY KEY,
turma_id integer,
FOREIGN KEY(turma_id) REFERENCES Turma (id)
);
